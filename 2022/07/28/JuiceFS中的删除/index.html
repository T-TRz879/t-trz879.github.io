JuiceFS中的删除

删除 回收站
JuiceFS

JuiceFS删除文件时会经历以下过程

  • 删除时判断文件是否被其它终端或用户打开，那么会调用延迟删除，将inode添加入removeFilesMap中，当其它人关闭文件时，会调用删除逻辑
  • 获取inode的每个c&inode_indx的数量，并对其进行判空操作，并对每个c&inode_indx进行删除
  • 获取c&inode_indx所有的slice并解析，对每次（100个slice/次）slice[]进行逐个删除
  • 每个slice中记录着chunkId和size，将待删除Block放入消息队列中
  • 消息队列消费删除消息时，按照如下删除顺序
    • 删除pendingKeysMap中待删除key
    • 删除客户端缓存
    • 删除对象存储中的数据
  • 等待数据删除完成后且没有错误，就会将c&inode_indx中当前chunk组合slice的Block删除

目前还未调研的场景

  • 延迟删除的过程中掉电/服务器挂了，后续会怎么办
    • 根据doUnlink的逻辑，检测到其它终端/用户正在打开文件时，删除不会将i&inode信息删除，且delayDelete调用的另外删除逻辑
    • 服务器掉电/挂了，在服务重启后，JuiceFS会有线程去查看哪些
  • 延迟删除的过程中其他终端/用户在写，新增数据会删除吗
    • A已经删除文件，但B在A删除之前打开文件并进行修改，B会保存成一个全新的文件

JuiceFS删除文件时涉及到的结构体和方法

baseMeta

  • removedFiles用来延迟删除的Map
  • engine当前使用的元数据引擎
type baseMeta struct {
    sync.Mutex         // 锁
    conf       *Config // 挂载目录的配置信息
    fmt        Format  // 存储当前format磁盘的信息

    root         Ino
    subTrash     internalNode
    sid          uint64
    of           *openfiles   // 缓存Attr和Slice
    removedFiles map[Ino]bool // 延迟删除Map
    compacting   map[uint64]bool
    deleting     chan int
    symlinks     *sync.Map
    msgCallbacks *msgCallbacks	// 消息队列
    newSpace     int64
    newInodes    int64
    usedSpace    int64 //当前用了多少空间
    usedInodes   int64 //当前Inode数量,也就是文件 + 文件夹数量
    umounting    bool  //是否是未挂载

    freeMu     sync.Mutex
    freeInodes freeID //用来对创建文件生成Inode的
    freeChunks freeID //用来对创建Chunk生成id的

    en engine	// 元数据引擎
}

baseMeta中删除Block的方法

rename和unlink时会调用此方法

  • opened：目前文件打开状态
  • inode：文件inode
  • length：文件长度
func (m *baseMeta) fileDeleted(opened bool, inode Ino, length uint64) {
    // length整个文件大小
    // 删除存储数据
    if opened {
        // 正在打开，目前无法删除
        m.Lock()
        m.removedFiles[inode] = true
        m.Unlock()
    } else {
        // 当前文件现在可以删除
        go m.en.doDeleteFileData(inode, length)
    }
}

engine接口中的删除方法

type engine interface{
    // 延迟删除
    doDeleteSustainedInode(sid uint64, inode Ino) error
    doFindDeletedFiles(ts int64, limit int) (map[Ino]uint64, error) // limit < 0 means all
    // 正常删除
    doDeleteFileData(inode Ino, length uint64)
    doCleanupSlices()
    doDeleteSlice(chunkid uint64, size uint32) error
}

注意：本文采用Redis作为元数据引擎

redisMeta.doDeleteFileData

利用redis的Pipeline每次获取1000条inode的c&inode_indx数据，直到indx*ChunkSize超过length。c&inode_indx中存放着inode当前Chunk的slice列表，slice中记录着每个Block的名称组成属性。

func (r *redisMeta) doDeleteFileData(inode Ino, length uint64) {
   logger.Infof("redisMeta doDeleteFileData ino:%d length:%d", inode, length)
   r.doDeleteFileData_(inode, length, "")
}

func (r *redisMeta) doDeleteFileData_(inode Ino, length uint64, tracking string) {
   var ctx = Background
   var indx uint32
   p := r.rdb.Pipeline()
   // 使用redis的pipeline 将所有的keys操作让如一个管道中传给redis进行操作，最后将结果返回
   for uint64(indx)*ChunkSize < length {
      var keys []string
      for i := 0; uint64(indx)*ChunkSize < length && i < 1000; i++ {
         // c&inode_indx
         key := r.chunkKey(inode, indx)
         keys = append(keys, key)
         _ = p.LLen(ctx, key)
         indx++
      }
      cmds, err := p.Exec(ctx)
      if err != nil {
         logger.Warnf("delete chunks of inode %d: %s", inode, err)
         return
      }
      // cmds是取出每个key的列表长度，也就是当前c&inode_indx中的block数
      for i, cmd := range cmds {
         val, err := cmd.(*redis.IntCmd).Result()
         if err == redis.Nil || val == 0 {
            continue
         }
         idx, _ := strconv.Atoi(strings.Split(keys[i], "_")[1])
         // 根据获取inode的block存放的信息，来对每一个Chunk进行删除
         err = r.deleteChunk(inode, uint32(idx))
         if err != nil {
            logger.Warnf("delete chunk %s: %s", keys[i], err)
            return
         }
      }
   }
   if tracking == "" {
      tracking = inode.String() + ":" + strconv.FormatInt(int64(length), 10)
   }
   _ = r.rdb.ZRem(ctx, delfiles, tracking)
}

redisMeta.deleteChunk

根据inode和indx可以定位具体是哪个c&inode_indx，进而取出当前第indx Chunk的slice列表，由于slice列表可能很多，每次只取100条，直到取完为止。

获取每个slice并解析放入slice[]中，解析的过程中就在删除slice，并遍历调用删除操作

func (r *redisMeta) deleteChunk(inode Ino, indx uint32) error {
   var ctx = Background
   key := r.chunkKey(inode, indx)
   for {
      // 用来保存所有待删除的slices
      var slices []*slice
      var rs []*redis.IntCmd
      err := r.txn(ctx, func(tx *redis.Tx) error {
         slices = nil
         // 获取c&inode_indx中 0-100的内容
         vals, err := tx.LRange(ctx, key, 0, 100).Result()
         if err == redis.Nil {
            return nil
         }
         _, err = tx.TxPipelined(ctx, func(pipe redis.Pipeliner) error {
            for _, v := range vals {
               rb := utils.ReadBuffer([]byte(v))
               _ = rb.Get32() // pos
               chunkid := rb.Get64()
               size := rb.Get32()
               slices = append(slices, &slice{chunkid: chunkid, size: size})
                // 删除c&inode_indx中的slice
               pipe.LPop(ctx, key)
               rs = append(rs, pipe.HIncrBy(ctx, sliceRefs, r.sliceKey(chunkid, size), -1))
            }
            return nil
         })
         return err
      }, key)
      if err != nil {
         return fmt.Errorf("delete slice from chunk %s fail: %s, retry later", key, err)
      }
      // 删除slices
      for i, s := range slices {
         if rs[i].Val() < 0 {
            r.deleteSlice(s.chunkid, s.size)
         }
      }
      // 如果小于100 则说明后面没有了
      if len(slices) < 100 {
         break
      }
   }
   return nil
}

baseMeta.deleteSlice

调用Msg来删除实体数据，实体数据删除成功后，删除slice的引用

func (m *baseMeta) deleteSlice(chunkid uint64, size uint32) {
   if m.conf.MaxDeletes == 0 {
      return
   }
   m.deleting <- 1
   defer func() { <-m.deleting }()
   // 删除本地缓存、对象存储中的数据
   err := m.newMsg(DeleteChunk, chunkid, size)
   if err != nil {
      logger.Warnf("delete chunk %d (%d bytes): %s", chunkid, size, err)
   } else {
      // 删除chunkId的sliceRef的引用
      err := m.en.doDeleteSlice(chunkid, size)
      if err != nil {
         logger.Errorf("delete slice %d: %s", chunkid, err)
      }
   }
}

registerMetaMsg

mount时会将DeleteChunk和CompactChunk两个消息队列初始化完成。

func registerMetaMsg(m meta.Meta, store chunk.ChunkStore, chunkConf *chunk.Config) {
    m.OnMsg(meta.DeleteChunk, func(args ...interface{}) error {
        return store.Remove(args[0].(uint64), int(args[1].(uint32)))
    })
    m.OnMsg(meta.CompactChunk, func(args ...interface{}) error {
        return vfs.Compact(*chunkConf, store, args[0].([]meta.Slice), args[1].(uint64))
    })
}

rChunk.Remove

根据c.length遍历删除所有的Block

func (c *rChunk) Remove() error {
   if c.length == 0 {
      // no block
      return nil
   }
   // 根据c.length计算有多少个Block
   lastIndx := (c.length - 1) / c.store.conf.BlockSize
   var err error
   for i := 0; i <= lastIndx; i++ {
      // c.id_i_len
      // there could be multiple clients try to remove the same chunk in the same time,
      // any of them should succeed if any blocks is removed
      key := c.key(i)
      c.store.pendingMutex.Lock()
      // 删除pending中本次待删除key
      delete(c.store.pendingKeys, key)
      c.store.pendingMutex.Unlock()
      // 删除本地缓存
      c.store.bcache.remove(key)
      // 删除远端的数据
      if e := c.delete(i); e != nil {
         err = e
      }
   }
   return err
}

延迟删除

baseMeta.Close

关闭文件时，会判断文件是否有延迟删除

func (m *baseMeta) Close(ctx Context, inode Ino) syscall.Errno {
   if m.of.Close(inode) {
      m.Lock()
      defer m.Unlock()
      if m.removedFiles[inode] {
         delete(m.removedFiles, inode)
         go func() {
            _ = m.en.doDeleteSustainedInode(m.sid, inode)
         }()
      }
   }
   return 0
}

redisMeta.doDeleteSustainedInode

首先删除inode的所有元数据，并更新totalInodes和useSapce

func (r *redisMeta) doDeleteSustainedInode(sid uint64, inode Ino) error {
   var attr Attr
   var ctx = Background
   a, err := r.rdb.Get(ctx, r.inodeKey(inode)).Bytes()
   if err == redis.Nil {
      return nil
   }
   if err != nil {
      return err
   }
   r.parseAttr(a, &attr)
   _, err = r.rdb.TxPipelined(ctx, func(pipe redis.Pipeliner) error {
       // 删除元数据
      pipe.ZAdd(ctx, delfiles, &redis.Z{Score: float64(time.Now().Unix()), Member: r.toDelete(inode, attr.Length)})
      pipe.Del(ctx, r.inodeKey(inode))
      pipe.IncrBy(ctx, usedSpace, -align4K(attr.Length))
      pipe.Decr(ctx, totalInodes)
      pipe.SRem(ctx, r.sustained(sid), strconv.Itoa(int(inode)))
      return nil
   })
   if err == nil {
       // 调用删除数据逻辑
      go r.doDeleteFileData(inode, attr.Length)
   }
   return err
}

JuiceFS中的回收站

一共有三种情况会涉及到回收站：

  • unlink删除文件
  • rmdir删除文件夹
  • rename重命名时dstFile存在

baseMeta.cleanupTrash

每小时执行一次，每次运行前都会检查元数据引擎中保存的lastCleanupTrash字段的值，在过去一小时内是否有更新过。

func (m *baseMeta) cleanupTrash() {
    for {
        // 每隔1小时执行一次
        utils.SleepWithJitter(time.Hour)
        if st := m.en.doGetAttr(Background, TrashInode, nil); st != 0 {
            if st != syscall.ENOENT {
                logger.Warnf("getattr inode %d: %s", TrashInode, st)
            }
            continue
        }
        // 校验上一次清除时间也是觉得需不需要进行清除 diff = 1hour
        if ok, err := m.en.setIfSmall("lastCleanupTrash", time.Now().Unix(), 3600); err != nil {
            logger.Warnf("checking counter lastCleanupTrash: %s", err)
        } else if ok {
            go m.doCleanupTrash(false)
        }
    }
}

baseMeta.doCleanupTrash

  • 获取.trash所有的时间文件夹
  • 比较是否达到删除期限
  • 遍历时间文件夹并获取所有的文件和文件夹并删除
  • 只有时间文件夹中所有内容全部清除成功才能将时间文件夹清除，否则放到下一次清除
    • 延迟删除、元数据操作尝试失败、元数据引擎错误等等原因
func (m *baseMeta) doCleanupTrash(force bool) {
   logger.Debugf("cleanup trash: started")
   ctx := Background
   now := time.Now()
   var st syscall.Errno
   var entries []*Entry
   if st = m.en.doReaddir(ctx, TrashInode, 0, &entries); st != 0 {
      logger.Warnf("readdir trash %d: %s", TrashInode, st)
      return
   }
   sort.Slice(entries, func(i, j int) bool { return entries[i].Inode < entries[j].Inode })
   var count int
   defer func() {
      if count > 0 {
         logger.Infof("cleanup trash: deleted %d files in %v", count, time.Since(now))
      }
   }()

   edge := now.Add(-time.Duration(24*m.fmt.TrashDays+1) * time.Hour)
   for _, e := range entries {
      ts, err := time.Parse("2006-01-02-15", string(e.Name))
      if err != nil {
         logger.Warnf("bad entry as a subTrash: %s", e.Name)
         continue
      }
      if ts.Before(edge) || force {
         var subEntries []*Entry
         if st = m.en.doReaddir(ctx, e.Inode, 0, &subEntries); st != 0 {
            logger.Warnf("readdir subTrash %d: %s", e.Inode, st)
            continue
         }
         // 其中是否有删除失败的情况
         rmdir := true
         for _, se := range subEntries {
            if se.Attr.Typ == TypeDirectory {
               // 删除文件夹
               st = m.en.doRmdir(ctx, e.Inode, string(se.Name))
            } else {
               // 其余的都是文件
               st = m.en.doUnlink(ctx, e.Inode, string(se.Name))
            }
            if st == 0 {
               count++
            } else {
               logger.Warnf("delete from trash %s/%s: %s", e.Name, se.Name, st)
               rmdir = false
               continue
            }
            if count%10000 == 0 && time.Since(now) > 50*time.Minute {
               return
            }
         }
         // 所有都删除成功后，将其回收站日期目录删除
         if rmdir {
            if st = m.en.doRmdir(ctx, TrashInode, string(e.Name)); st != 0 {
               logger.Warnf("rmdir subTrash %s: %s", e.Name, st)
            }
         }
      } else {
         break
      }
   }
}
